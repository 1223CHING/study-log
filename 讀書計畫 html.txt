        // Call Gemini API to generate daily tasks
        async function generateAITasks() {
            const subjectInput = document.getElementById('ai-modal-subject-input').value.trim();
            const intensity = document.getElementById('ai-modal-intensity').value;

            if (!subjectInput) {
                showNotification("請輸入你今天想要讀的科目或章節範圍！📚", "fa-solid fa-circle-exclamation text-rose-400");
                return;
            }

            // UI Transition to Loading State
            document.getElementById('ai-form-fields').classList.add('hidden');
            document.getElementById('ai-modal-buttons').classList.add('hidden');
            const loadingContainer = document.getElementById('ai-loading-container');
            const loadingText = document.getElementById('ai-loading-text');
            loadingContainer.classList.remove('hidden');

            // Dynamic loading message rotations
            let loadingInterval = setInterval(() => {
                const randomLine = AI_LOADING_LINES[Math.floor(Math.random() * AI_LOADING_LINES.length)];
                loadingText.innerText = randomLine;
            }, 2000);

            // Construct LLM Prompts & Structured Query
            const systemPrompt = "你是一位充滿熱忱、有智慧的台灣高中導師。你要幫助學生將他們當天想要研讀的廣泛章節，拆解成3-5個「具體、好落實、具備實操性」的讀書目標清單。字句必須精簡扼要，口氣溫暖和煦且能帶來正向激勵。";
            
            const userQuery = `
                【學生的終極大考目標】：${appState.ultimateGoal}
                【今天想攻克或複習的單元/科目】：${subjectInput}
                【期望的學習規劃強度風格】：${intensity}
                
                請為這位同學規劃 3 到 5 個具體的、今天內可落實的細部讀書計畫目標。
                例如，如果他想讀「三角函數」，不要寫「讀完三角函數」，而要幫他拆解為「複弦定理公式並理解原理」、「練習講義基礎題 5 題」等極易執行的步驟。
                每個規劃請保持精簡（25字以內），請完全以「繁體中文（台灣繁體）」格式回答。
            `;

            const apiKey = ""; // Canvas Runtime will automatically inject this
            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-3-flash-preview:generateContent?key=${apiKey}`;

            const payload = {
                contents: [{ parts: [{ text: userQuery }] }],
                generationConfig: {
                    responseMimeType: "application/json",
                    responseSchema: {
                        type: "ARRAY",
                        description: "每日具體讀書計畫任務清單",
                        items: {
                            type: "OBJECT",
                            properties: {
                                "taskName": { 
                                    "type": "STRING",
                                    "description": "具體易執行的每日學習小任務目標" 
                                }
                            },
                            "required": ["taskName"]
                        }
                    }
                },
                systemInstruction: {
                    parts: [{ text: systemPrompt }]
                }
            };

            // Implement Exponential Backoff for API stability
            let attempts = 0;
            const maxAttempts = 3;
            let responseReceived = false;
            let resultJSON = null;

            while (attempts < maxAttempts && !responseReceived) {
                try {
                    const response = await fetch(apiUrl, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });

                    if (response.ok) {
                        const result = await response.json();
                        const candidate = result.candidates?.[0];
                        if (candidate && candidate.content?.parts?.[0]?.text) {
                            resultJSON = JSON.parse(candidate.content.parts[0].text);
                            responseReceived = true;
                        }
                    }
                } catch (error) {
                    console.warn(`Gemini API 嘗試次數 ${attempts + 1} 失敗：`, error);
                }

                if (!responseReceived) {
                    attempts++;
                    if (attempts < maxAttempts) {
                        await new Promise(resolve => setTimeout(resolve, Math.pow(2, attempts) * 1000));
                    }
                }
            }

            clearInterval(loadingInterval);

            // Process response
            if (responseReceived && resultJSON && Array.isArray(resultJSON) && resultJSON.length > 0) {
                // Add generated tasks to current todos
                resultJSON.forEach(item => {
                    if (item.taskName && item.taskName.trim()) {
                        const newId = appState.todos.length > 0 ? Math.max(...appState.todos.map(t => t.id)) + 1 : 1;
                        appState.todos.push({
                            id: newId,
                            text: item.taskName.trim(),
                            completed: false
                        });
                    }
                });

                saveToLocalStorage();
                updateTodos();
                closeAIModal();
                showNotification("AI 導師成功為你量身打造了學習計畫！衝刺吧！🚀", "fa-solid fa-wand-magic-sparkles text-purple-400");
            } else {
                // Failback / Error feedback
                document.getElementById('ai-form-fields').classList.remove('hidden');
                document.getElementById('ai-modal-buttons').classList.remove('hidden');
                loadingContainer.classList.add('hidden');
                showNotification("AI 規劃時出了點小差錯，請再點擊一次重試看看！🔌", "fa-solid fa-triangle-exclamation text-rose-500");
            }
        }

                const newId = appState.todos.length > 0 ? Math.max(...appState.todos.map(t => t.id)) + 1 : 1;
                appState.todos.push({
                    id: newId,
                    text: text,
                    completed: false
                });
                input.value = '';
                saveToLocalStorage();
                updateTodos();
            }
        }

        // Toggle to-do item completed
        function toggleTodo(id) {
            const todo = appState.todos.find(t => t.id === id);
            if (todo) {
                const wasCompleted = todo.completed;
                todo.completed = !todo.completed;
                saveToLocalStorage();
                updateTodos();

                // If all completed right after this action, trigger fireworks
                const completedCount = appState.todos.filter(t => t.completed).length;
                if (!wasCompleted && completedCount === appState.todos.length && appState.todos.length > 0) {
                    triggerConfetti();
                    showNotification("太厲害了！今日讀書目標全數完成！🏆", "fa-solid fa-trophy text-yellow-400");
                }
            }
        }

        // Delete to-do item
        function deleteTodo(id) {
            appState.todos = appState.todos.filter(t => t.id !== id);
            saveToLocalStorage();
            updateTodos();
        }

        // Mood Selection Handling
        function selectMood(moodKey, emoji) {
            // Remove active classes from all buttons
            document.querySelectorAll('.mood-btn').forEach(btn => btn.classList.remove('active'));
            // Set state
            appState.selectedMood = moodKey;
            appState.selectedMoodEmoji = emoji;
            // Activate button
            const activeBtn = document.getElementById(`mood-${moodKey}`);
            if (activeBtn) activeBtn.classList.add('active');
        }

        // Archive Day Logic
        function handleArchiveProgress() {
            const totalCount = appState.todos.length;
            const completedCount = appState.todos.filter(t => t.completed).length;
            const reflectionText = document.getElementById('reflection-input').value.trim();

            if (totalCount === 0) {
                showNotification("今天還沒設定任何讀書目標，沒辦法歸檔喔！", "fa-solid fa-circle-exclamation text-amber-500");
                return;
            }

            const now = new Date();
            const dateStr = `${now.getMonth() + 1}/${now.getDate()}`;

            // Add new history record, storing the actual tasks detailed data!
            const newHistoryItem = {
                date: dateStr,
                completedCount: completedCount,
                totalCount: totalCount,
                mood: appState.selectedMood,
                moodEmoji: appState.selectedMoodEmoji,
                reflection: reflectionText,
                tasks: appState.todos.map(t => ({ text: t.text, completed: t.completed }))
            };

            appState.history.push(newHistoryItem);
            
            // Clean up today's panel
            appState.todos = [];
            document.getElementById('reflection-input').value = '';
            appState.selectedMood = "fighting";
            appState.selectedMoodEmoji = "🔥";
            selectMood("fighting", "🔥");

            saveToLocalStorage();
            updateTodos();
            updateHistoryFootprint();
            
            // Trigger beautiful archiving celebratory feedback
            triggerConfetti();
            showNotification("今日讀書足跡歸檔成功！祝明天也順利！🚀", "fa-solid fa-circle-check text-emerald-400");
        }

            if (appState.history.length === 0) return;

        }

        // HTML escaping utility
        function escapeHtml(str) {
            return str
                .replace(/&/g, "&amp;")
                .replace(/</g, "&lt;")
                .replace(/>/g, "&gt;")
                .replace(/"/g, "&quot;")
                .replace(/'/g, "&#039;");
        }

        // Custom self-contained Canvas Confetti Engine (100% reliable)
        let canvas, ctx;
        let confetti = [];
        const confettiCount = 120;
        const gravity = 0.5;
        const terminalVelocity = 5;
        const drag = 0.075;
        const colors = [
            { front: '#4f46e5', back: '#4338ca' }, // Indigo
            { front: '#10b981', back: '#059669' }, // Emerald
            { front: '#f59e0b', back: '#d97706' }, // Amber
            { front: '#ec4899', back: '#db2777' }, // Pink
            { front: '#3b82f6', back: '#2563eb' }  // Blue
        ];

        function initConfetti() {
            canvas = document.getElementById('confetti-canvas');
            ctx = canvas.getContext('2d');
            resizeCanvas();
            window.addEventListener('resize', resizeCanvas);
        }

        function resizeCanvas() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }

        function randomRange(min, max) {
            return Math.random() * (max - min) + min;
        }

        function createConfettiParticle() {
            return {
                color: colors[Math.floor(Math.random() * colors.length)],
                dimensions: {
                    x: randomRange(10, 20),
                    y: randomRange(10, 30)
                },
                position: {
                    x: randomRange(0, canvas.width),
                    y: canvas.height - 10 // shoot from bottom
                },
                rotation: randomRange(0, 2 * Math.PI),
                scale: {
                    x: 1,
                    y: 1
                },
                velocity: {
                    x: randomRange(-25, 25),
                    y: randomRange(-20, -35) // intense burst upwards
                }
            };
        }

        let isRunning = false;
        function triggerConfetti() {
            confetti = [];
            for (let i = 0; i < confettiCount; i++) {
                confetti.push(createConfettiParticle());
            }
            if (!isRunning) {
                isRunning = true;
                animateConfetti();
            }
        }

        function animateConfetti() {
            if (confetti.length === 0) {
                isRunning = false;
                ctx.clearRect(0, 0, canvas.width, canvas.height);
                return;
            }

            ctx.clearRect(0, 0, canvas.width, canvas.height);

            confetti.forEach((particle, index) => {
                particle.position.x += particle.velocity.x;
                particle.position.y += particle.velocity.y;
                particle.velocity.x -= particle.velocity.x * drag;
                particle.velocity.y += gravity;
                particle.rotation += particle.velocity.x * 0.05;

                // draw piece
                ctx.save();
                ctx.translate(particle.position.x, particle.position.y);
                ctx.rotate(particle.rotation);
                
                ctx.scale(particle.scale.x, particle.scale.y);
                ctx.fillStyle = particle.color.front;
                ctx.fillRect(-particle.dimensions.x / 2, -particle.dimensions.y / 2, particle.dimensions.x, particle.dimensions.y);
                ctx.restore();

                // delete particles falling out of frame
                if (particle.position.y > canvas.height) {
                    confetti.splice(index, 1);
                }
            });

            requestAnimationFrame(animateConfetti);
        }
